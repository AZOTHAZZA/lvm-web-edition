<script>
    // --- 記憶層 (LogosStorage) ---
    window.LogosStorage = {
        save: (key, data) => localStorage.setItem(`LVM_${key}`, JSON.stringify(data)),
        load: (key) => {
            const data = localStorage.getItem(`LVM_${key}`);
            return data ? JSON.parse(data) : null;
        },
        getAssets: function() { return this.load('ASSETS') || {}; },
        registerAsset: function(symbol, details) {
            const assets = this.getAssets();
            assets[symbol] = { symbol, balance: details.balance || '0' };
            this.save('ASSETS', assets);
            console.log("STORAGE: REGISTERED " + symbol);
        }
    };

    // --- 司令塔ロジック (lvm_bridge_logic) ---
    window.LVM_BRIDGE = {
        version: "2.4.2",
        processCommand: function(input) {
            const args = input.trim().toUpperCase().split(/\s+/);
            const cmd = args[0];

            if (cmd === "GENERATE") {
                if (args[2]) {
                    window.LogosStorage.registerAsset(args[2], { balance: "0" });
                    window.sysLog(`SUCCESS: ${args[2]}_GENERATED`, "system");
                } else {
                    window.sysLog("USAGE: GENERATE CURRENCY [SYM]", "error");
                }
            } else if (cmd === "MINT") {
                window.sysLog(`MINT_PROTOCOL_EXECUTED: ${args[1]} ${args[2]}`, "system");
                // 数理処理が必要な場合は LogosDecimal を使用
            } else {
                window.sysLog(`UNKNOWN_COMMAND: ${cmd}`, "error");
            }
        }
    };
    console.log("LVM_BRIDGE: INTEGRATED_INTO_HTML");
</script>

<script>
    const output = document.getElementById('output');
    const statusElement = document.getElementById('system-status');

    window.sysLog = function(msg, type = '') {
        const div = document.createElement('div');
        div.className = type;
        div.innerHTML = `> ${msg}`;
        output.appendChild(div);
        output.scrollTop = output.scrollHeight;

        if (msg.includes("READY")) {
            statusElement.innerText = "READY";
            statusElement.className = "ready";
        }
    };

    const inputField = document.getElementById('user-input');
    inputField.addEventListener('keypress', (e) => {
        if (e.key === 'Enter') {
            const cmd = inputField.value;
            output.innerHTML += `<br><span style="color:#fff">USER@LVM:~$ ${cmd}</span>`;
            
            if (window.LVM_BRIDGE) {
                window.LVM_BRIDGE.processCommand(cmd);
            } else {
                window.sysLog("ERROR: CORE_NOT_READY", "error");
            }
            inputField.value = '';
            output.scrollTop = output.scrollHeight;
        }
    });

    // 起動シーケンス
    window.addEventListener('load', () => {
        setTimeout(() => {
            window.sysLog("LVM_CORE: ALL_SYSTEMS_OPERATIONAL_READY", "system");
        }, 800);
    });
</script>
