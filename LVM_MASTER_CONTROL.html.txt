<script>
    // システムステータスを更新する関数
    function updateSystemStatus(status) {
        const statusElement = document.getElementById('system-status');
        const output = document.getElementById('output');
        
        statusElement.innerText = status;
        if (status === 'READY') {
            statusElement.className = 'ready';
            output.innerHTML += `<br><span style="color:#00ff00">> LVM_CORE_v2.2: CONNECTION_ESTABLISHED.</span>`;
            output.innerHTML += `<br>> ALL SYSTEMS OPERATIONAL.`;
        }
    }

    // 以前の Service Worker 登録処理は、キャッシュエラー防止のため完全に削除しました。

    // コマンド入力イベント
    const inputField = document.getElementById('user-input');
    inputField.addEventListener('keypress', async (e) => {
        if (e.key === 'Enter') {
            const cmd = inputField.value;
            const output = document.getElementById('output');
            output.innerHTML += `<br><span style="color:#fff">USER@LVM:~$ ${cmd}</span>`;
            inputField.value = '';

            // Bridge経由でメッセージ送信
            if (window.lvmBridge) {
                const response = await window.lvmBridge.sendMessage(cmd);
                output.innerHTML += `<br>> ${response}`;
            }
            output.scrollTop = output.scrollHeight;
        }
    });

    // 起動時の初期化ログ
    window.addEventListener('load', () => {
        console.log("LVM_UI: INTERFACE_LOADED");
    });
</script>
